cmake_minimum_required(VERSION 3.12)
project(AsyncTaskFlow VERSION 1.0.0)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)


add_executable(AsyncTaskFlow src/main.cpp)

# Include directories
target_include_directories(AsyncTaskFlow PRIVATE include)

# Install target
install(TARGETS AsyncTaskFlow DESTINATION bin)

if(EXISTS ${CMAKE_CURRENT_SOURCE_DIR}/test)
    enable_testing()
    add_subdirectory(test)
endif()
